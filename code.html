<script type="text/javascript">
var mfrColors = d3.scaleOrdinal()
.domain(['Drama','Science Fiction','Comedy','Music','Mystery','Action','War','Horror','Westerns'])
.range(["#8dd3c7","#ffffb3","#bebada","#fb8072","#80b1d3",
"#fdb462","#b3de69","#d66fa4","#bc80bd","#ccebc5","#ffed6f"])
var awardscolor = d3.scaleOrdinal()
.domain(['Yes','No'])
.range(["#97BC62FF", "#F08080"])
var chart = dc.boxPlot("#boxtest"),
pie1 = new dc.PieChart("#piechart"),
pie2 = new dc.PieChart("#pie2");
d3.csv("films.csv").then(function(experiments) {
experiments.forEach(function(x) {
x.Speed = +x.Speed;
});
var ndx = crossfilter(experiments),
awardsDimension = ndx.dimension(function(d) {return d.Run;}),
awardsGroup = awardsDimension.group(),
subjectDim = ndx.dimension(function(d) {return d.Expt;}),
subjectGroup = subjectDim.group().reduceCount(function(d) {return d.Title;}),
subjectDimension = ndx.dimension(function(d) {return d.Expt;}),
subjectArrayGroup = subjectDimension.group().reduce(
function(p,v) {
// keep array sorted for efficiency
p.splice(d3.bisectLeft(p, v.Speed), 0, v.Speed);
return p;
},
function(p,v) {
p.splice(d3.bisectLeft(p, v.Speed), 1);
return p;
},
function() {
return [];
}
);
d3.select('#resetButton')
.on('click', function() {
dc.filterAll();
dc.redrawAll();
dc.filterAll();
dc.redrawAll();
});
pie1
.width(300)
.height(300)
.innerRadius(50)
.radius(160)
.dimension(awardsDimension)
.group(awardsGroup)
.colors(awardscolor)
.controlsUseVisibility(true)
chart
.width(900)
.height(600)
.margins({top: 50, right: 50, bottom: 30, left: 30})
.dimension(subjectDimension)
.group(subjectArrayGroup)
.elasticY(true)
.elasticX(true)
.xAxisLabel('Subject')
.yAxisLabel('Popularity')
.controlsUseVisibility(true)
.colors(mfrColors);
pie2
.width(500)
.height(300)
.radius(140)
.dimension(subjectDim)
.group(subjectGroup)
.controlsUseVisibility(true)
.colors(mfrColors)
.legend(dc.legend().x(0).y(60).itemHeight(13).gap(5).horizontal(1).legendWidth(10).itemWidth(70));
var fil = new dc.BarChart('#testings');
typeDimension = ndx.dimension(function(d) {return d.Expt;}),
typeGroup = typeDimension.group().reduceCount(function(d) {return +d.Length;});
fil
.width(768)
.height(400)
.x(d3.scaleBand())
.xUnits(dc.units.ordinal)
.brushOn(true)
.xAxisLabel('Subject')
.yAxisLabel('Length')
.dimension(typeDimension)
.group(typeGroup)
.elasticX(true)
.barPadding(0.3)
.controlsUseVisibility(true)
.colors(mfrColors)
.colorAccessor(function(d) {
return d.key
});
var focusChart = new dc.SeriesChart("#serees");
var sumofLengthDim = ndx.dimension(function(d) {return [d.Expt, +d.Year]; });
var sumofLengthGroup = sumofLengthDim.group().reduceSum(function(d) { return +d.Length; });
focusChart
.transitionDuration(0)
.chart(function(c) { return new dc.LineChart(c).curve(d3.curveCardinal).evadeDomainFilter(true); })
.x(d3.scaleLinear().domain([1920,1995]))
.controlsUseVisibility(true)
.brushOn(false)
.yAxisLabel("Length of films")
.yAxisPadding("10%")
.xAxisLabel("Years")
.elasticY(true)
.dimension(sumofLengthDim)
.group(sumofLengthGroup)
.seriesAccessor(function(d) {return d.key[0];})
.keyAccessor(function(d) {return +d.key[1];})
.valueAccessor(function(d) {return +d.value ;})
.colors(mfrColors)
.legend(dc.legend().x(200).y(100).itemHeight(13).gap(5).horizontal(1).legendWidth(10).itemWidth(70));
focusChart.yAxis().tickFormat(function(d) {return d3.format(',d')(d);});
focusChart.margins().left += 40;
dc.renderAll();
});
